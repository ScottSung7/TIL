```
1. yum 이용
sudo yum update 

// 오라클 JDK
sudo yum install java-11-openjdk java-11-openjdk-devel

//환경변수 - 모든 유저
sudo vi /etc/profile

export JAVA_HOME=$(dirname $(dirname $(readlink -f $(which javac))))
export PATH=$PATH:$JAVA_HOME/bin

source /etc/profile

//환경변수 - 특정 유저
vi ~/.bashrc

export JAVA_HOME=$(dirname $(dirname $(readlink -f $(which javac))))
export PATH=$PATH:$JAVA_HOME/bin

source ~/.
bashrc

2. homebrew 이용
brew update
brew install openjdk@21

```
